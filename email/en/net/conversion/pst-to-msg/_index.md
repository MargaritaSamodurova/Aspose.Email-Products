using (var pst = Aspose.Email.Storage.Pst.PersonalStorage.FromFile("myStorage.pst")){
// select "Inbox" folder in PST
var inbox = pst.RootFolder.GetSubFolder("Inbox");
// enumerate all messages in "Inbox" folder
foreach (var messageInfo in inbox.EnumerateMessages()){
    // create FileStream to save messages
    using (FileStream stream = File.OpenWrite(messageInfo.Subject + ".msg")){
         // save message in MSG format and write to disc
        pst.SaveMessageToStream(messageInfo.EntryIdString, stream);
    }
}
}

MSG is a file format used by Microsoft Outlook and Exchange to store email messages, contact, appointment, or other tasks. Such messages may contain one or more email fields, with the sender, recipient, subject, date, and message body, or contact information, appointment particulars, and one or more task specifications.
